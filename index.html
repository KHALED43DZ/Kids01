// ====== الاتصال بـ Flowise/LangFlow ======
// ضع رابط Flowise الخاص بك أدناه

const FLOWISE_URL = 'http://localhost:3000/api/v1/prediction/YOUR_FLOW_ID';

async function callFlowiseWorkflow(topic) {
  const response = await fetch(FLOWISE_URL, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      question: topic,
      // تمرير الإعدادات للـ Flow
      overrideConfig: {
        questionCount: 5,
        ageGroup: '6-8',
        imageStyle: 'pixar',
        dialect: 'msa'
      }
    })
  });
  
  const data = await response.json();
  return data.text; // JSON string من LLM
}

// ====== Prompt موصى به لـ System Message ======
